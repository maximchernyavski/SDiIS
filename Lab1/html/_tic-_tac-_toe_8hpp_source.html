246 int check_direction_combination(int x, int y, char dir, int delta, int prev_x, int prev_y) {
247 std::vector<int> signs = get_direction_signs(dir);
248
249 if (field_[x + y * num_of_columns_] == current_symbol_ && x >= 0 && y >= 0 && x < num_of_columns_ && y < num_of_rows_) {
250 int new_x1 = x + 1 * signs[0];
251 int new_y1 = y + 1 * signs[1];
252 int new_x2 = x + 1 * signs[2];
253 int new_y2 = y + 1 * signs[3];
254
255 if (!(new_x1 == prev_x && new_y1 == prev_y)) {
256 delta += check_direction_combination(new_x1, new_y1, dir, delta, x, y);
257 }
258 if (!(new_x2 == prev_x && new_y2 == prev_y)) {
259 delta += check_direction_combination(new_x2, new_y2, dir, delta, x, y);
260 }
261 }
